## 一、项目概述

### 1.1 优化背景

基于对美国主流宠物健康应用（Rover、PetDesk、Chewy）的分析，当前应用需要以下优化以提升美国用户体验：

1. **导航结构优化** - 从纯层级导航改为底部标签栏导航
2. **今日视图缺失** - 添加 Today 视图显示当日任务
3. **设置入口调整** - 设置入口移至底部导航或右上角
4. **删除确认增强** - 重要数据删除需更明确警示

### 1.2 优化目标

| 目标 | 当前状态 | 目标状态 | 衡量标准 |
|------|---------|---------|---------|
| 导航效率 | 3 层点击到达设置 | 1 层点击 | 点击次数减少 66% |
| 提醒可见性 | 需进入宠物详情查看 | 首页直接显示 | 提醒触达率提升 |
| 删除安全性 | 简单确认 | 详细警示 | 误删率降低 |

### 1.3 设计原则

| 原则 | 说明 | 实施方式 |
|------|------|---------|
| 单一职责 | 一个功能一个模块 | 每个视图文件只负责一个界面 |
| 高内聚低耦合 | 相关功能聚合 | ViewModel 集中管理业务逻辑 |
| 命名清晰 | 语义化命名 | `TodayView`而非`NewView` |
| 文件结构语义化 | 目录即文档 | `Views/`、`Components/`、`ViewModels/` |
| 代码复用 | 优先复用现有模块 | 抽象符合"三次法则" |
| 最小代码 | 避免过度设计 | 用最简单方案解决问题 |
| 编译验证 | 每次改动后验证 | xcodebuild 自动编译 |
| 清理废弃 | 标记→验证→删除 | 提交信息注明清理范围 |

### 1.4 与现有代码切合点

| 现有模块 | 复用方式 | 扩展内容 |
|----------|---------|---------|
| `PetListView` | 改为 Home Tab 内容 | 添加设置入口 |
| `PetDetailView` | 保持不变 | 仅调整导航入口 |
| `NotificationService` | 复用调度逻辑 | Today 视图数据源 |
| `SharedDataManager` | 复用数据管理 | 无修改 |
| `SwiftData` 模型 | 完全复用 | 无修改 |

// 文件：pethealthdata/pethealthdata/ViewModels/TodayViewModel.swift

import Foundation
import SwiftData
import Combine

/// 今日提醒视图模型
@MainActor
class TodayViewModel: ObservableObject {
    @Published var todayVaccines: [VaccineRecord] = []
    @Published var todayMedications: [Medication] = []
    @Published var upcomingReminders: [UpcomingReminder] = []
    @Published var isLoading: Bool = false
    @Published var errorMessage: String?
    
    private let modelContext: ModelContext
    private var cancellables = Set<AnyCancellable>()
    
    init(modelContext: ModelContext) {
        self.modelContext = modelContext
    }
    
    /// 加载今日提醒
    func loadTodaysReminders() {
        isLoading = true
        errorMessage = nil
        
        do {
            let calendar = Calendar.current
            let today = calendar.startOfDay(for: Date())
            let endOfToday = calendar.date(byAdding: .day, value: 1, to: today) ?? today
            let threeDaysFromNow = calendar.date(byAdding: .day, value: 3, to: today) ?? today
            
            // 加载今日疫苗
            let vaccineDescriptor = FetchDescriptor<VaccineRecord>(
                predicate: #Predicate { vaccine in
                    guard let nextDueDate = vaccine.nextDueDate else { return false }
                    return nextDueDate >= today && nextDueDate < endOfToday
                }
            )
            todayVaccines = try modelContext.fetch(vaccineDescriptor)
            
            // 加载今日药物
            let medicationDescriptor = FetchDescriptor<Medation>(
                predicate: #Predicate { medication in
                    return medication.isActive
                }
            )
            let allMedications = try modelContext.fetch(medicationDescriptor)
            todayMedications = allMedications.filter { medication in
                guard let startDate = medication.startDate as? Date else { return false }
                guard startDate <= today else { return false }
                
                if let endDate = medication.endDate {
                    guard endDate >= today else { return false }
                }
                
                // 根据频率判断今日是否需要服用
                return medication.isDueToday(today: today)
            }
            
            // 加载即将到来的提醒
            upcomingReminders = loadUpcomingReminders(until: threeDaysFromNow)
            
            isLoading = false
        } catch {
            errorMessage = error.localizedDescription

        }

                        .font(.system(size: 16))

            }

            VStack(alignment: .leading, spacing: 4) {

                        .font(.system(size: 13))
                        .foregroundColor(.appTextSecondary)
