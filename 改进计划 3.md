### 1.1 问题背景

根据深度分析，当前应用存在三大核心问题：

1. **通知提醒可靠性问题**：缺少授权检查、时间计算缺陷、声音文件验证缺失
2. **Today 视图显示准确性问题**：日期匹配逻辑错误、缺少已完成过滤、性能问题
3. **声音选择交互问题**：缺少视觉反馈、预览功能不完善、交互体验差

**注意**：原计划中提到的 SettingsView 的 `.constant` 问题已重新评估，确认现有使用方式正确，无需修改。

### 1.2 优化目标

- ✅ **可靠性**：确保通知提醒 100% 可靠，有完善的错误处理

1. **单一职责**：一个功能一个模块，高内聚低耦合
2. **代码复用**：优先复用现有模块，抽象符合"三次法则"

### 模块一：通知权限服务

#### 3.1.1 需求描述

**问题**：当前通知调度前未检查授权状态，导致提醒静默失败。

**需求**：
- 创建独立的通知权限服务，统一管理授权状态
- 提供授权状态检查、请求授权、状态监听功能
- 在调度通知前必须验证授权

**文件**：`pethealthdata/pethealthdata/Services/NotificationPermissionService.swift`

#### 3.1.2 生成规则

**代码结构**：

```swift
import Foundation
import UserNotifications

/// 通知权限管理服务
/// 
/// 负责管理通知授权状态，提供授权检查、请求和监听功能。
/// 所有通知调度前必须通过此服务验证授权。
final class NotificationPermissionService {
    static let shared = NotificationPermissionService()
    
    // MARK: - Properties
    
    /// 当前授权状态
    private var cachedStatus: UNAuthorizationStatus?
    
    /// 授权状态变化监听器
    private var statusListeners: [(UNAuthorizationStatus) -> Void] = []
    
    private init() {
        // 初始化时缓存状态
        refreshStatus()
    }
    
    // MARK: - Public Methods
    
    /// 刷新授权状态
    func refreshStatus() {
        UNUserNotificationCenter.current().getNotificationSettings { [weak self] settings in
            DispatchQueue.main.async {
                self?.cachedStatus = settings.authorizationStatus
                self?.notifyListeners(settings.authorizationStatus)
            }
        }
    }
    
    /// 获取当前授权状态
    /// - Returns: 当前授权状态
    var authorizationStatus: UNAuthorizationStatus {
        return cachedStatus ?? .notDetermined
    }
    
    /// 检查是否已授权
    /// - Returns: true 表示已授权，可以发送通知
    func isAuthorized() -> Bool {
        guard let status = cachedStatus else { return false }
        
        switch status {
        case .authorized, .provisional:
            return true
        case .denied, .ephemeral, .notDetermined:
            return false
        @unknown default:
            return false
        }
    }
    
    /// 请求通知授权
    /// - Parameter completion: 完成回调，参数为是否授权成功
    func requestAuthorization(completion: @escaping (Bool) -> Void) {
        UNUserNotificationCenter.current().requestAuthorization(
            options: [.alert, .badge, .sound, .criticalAlert]
        ) { [weak self] granted, error in
            DispatchQueue.main.async {
                self?.refreshStatus()
                
                if let error = error {
                    print("❌ Notification authorization error: \(error)")
                    completion(false)
                } else {
                    print("✅ Notification permission \(granted ? "granted" : "denied")")
                    completion(granted)
                }
            }
        }
    }
    
    /// 检查并请求授权（如果需要）
    /// - Parameter completion: 完成回调
    func checkAndRequestIfNeeded(completion: @escaping (Bool) -> Void) {
        if isAuthorized() {
            completion(true)
        } else {
            requestAuthorization(completion: completion)
        }
    }
    
    /// 添加授权状态监听器
    /// - Parameter listener: 状态变化回调
    func addStatusListener(_ listener: @escaping (UNAuthorizationStatus) -> Void) {
        statusListeners.append(listener)
        // 立即触发一次当前状态
        if let status = cachedStatus {
            listener(status)
        }
    }
    
    /// 移除授权状态监听器
    /// - Parameter listener: 要移除的监听器
    func removeStatusListener(_ listener: @escaping (UNAuthorizationStatus) -> Void) {
        statusListeners.removeAll { whereClause in
            // 无法直接比较闭包，这里简化处理
            return false
        }
    }
    
    // MARK: - Private Methods
    
    /// 通知所有监听器状态变化
    private func notifyListeners(_ status: UNAuthorizationStatus) {
        for listener in statusListeners {
            listener(status)
        }
    }
}
```

**代码说明**：
- 使用单例模式，全局统一管理
- 提供缓存机制，避免频繁查询
- 支持状态监听，便于 UI 实时更新
- 所有方法都在主线程回调

#### 3.2.2 生成规则

**代码结构**：

```swift
import Foundation
import AVFoundation

/// 通知声音验证服务
/// 
/// 负责验证声音文件是否存在，提供预览播放和降级方案。
/// 合并了原 AudioPreviewService 的功能。
final class NotificationSoundValidator {
    static let shared = NotificationSoundValidator()
    
    // MARK: - Properties
    
    /// 音频播放器
    private var audioPlayer: AVAudioPlayer?
    
    /// 默认声音文件名（不含扩展名）
    private let defaultSoundName = "triTone"
    
    private init() {}
    
    // MARK: - Public Methods
    
    /// 验证声音文件是否存在
    /// - Parameter fileName: 声音文件名（含扩展名，如 "triTone.caf"）
    /// - Returns: true 表示文件存在
    func validateSoundFile(fileName: String) -> Bool {
        // Silent 选项没有文件
        guard !fileName.isEmpty else { return true }
        
        let soundName = String(fileName.dropLast(4)) // 移除 .caf
        return Bundle.main.url(forResource: soundName, withExtension: "caf") != nil
    }
    
    /// 获取声音文件 URL
    /// - Parameter fileName: 声音文件名
    /// - Returns: 文件 URL，不存在则返回 nil
    func soundURL(for fileName: String) -> URL? {
        guard !fileName.isEmpty else { return nil }
        
        let soundName = String(fileName.dropLast(4))
        return Bundle.main.url(forResource: soundName, withExtension: "caf")
    }
    
    /// 播放预览声音
    /// - Parameters:
    ///   - fileName: 声音文件名
    ///   - completion: 播放完成回调
    func playPreview(fileName: String, completion: (() -> Void)? = nil) {
        guard !fileName.isEmpty else {
            print("⚠️ Cannot play silent sound")
            completion?()
            return
        }
        
        guard let url = soundURL(for: fileName) else {
            print("⚠️ Sound file not found: \(fileName), falling back to default")
            // 降级到默认声音
            playDefaultSound(completion: completion)
            return
        }
        
        do {
            audioPlayer = try AVAudioPlayer(contentsOf: url)
            audioPlayer?.prepareToPlay()
            audioPlayer?.play()
            
            // 播放完成后回调
            audioPlayer?.delegate = self
            if completion != nil {
                // 存储回调到代理（简化处理，实际可使用闭包）
            }
        } catch {
            print("❌ Failed to play sound: \(error)")
            playDefaultSound(completion: completion)
        }
    }
    
    /// 停止播放
    func stopPreview() {
        audioPlayer?.stop()
        audioPlayer = nil
    }
    
    /// 获取用于通知的内容声音
    /// - Parameter fileName: 用户选择的声音文件名
    /// - Returns: UNNotificationSound 对象（保证可用）
    func notificationSound(for fileName: String) -> UNNotificationSound {
        guard !fileName.isEmpty, fileName != "silent" else {
            return .default
        }
        
        guard validateSoundFile(fileName: fileName),
              let url = soundURL(for: fileName) else {
            // 文件不存在，降级到默认
            return .default
        }
        
        return UNNotificationSound(named: UNNotificationSoundName(url.lastPathComponent))
    }
    
    // MARK: - Private Methods
    
    /// 播放默认声音
    private func playDefaultSound(completion: (() -> Void)?) {
        guard let url = Bundle.main.url(forResource: defaultSoundName, withExtension: "caf") else {
            completion?()
            return
        }
        
        do {
            audioPlayer = try AVAudioPlayer(contentsOf: url)
            audioPlayer?.play()
        } catch {
            print("❌ Failed to play default sound: \(error)")
        }
    }
}

// MARK: - AVAudioPlayerDelegate

extension NotificationSoundValidator: AVAudioPlayerDelegate {
    func audioPlayerDidFinishPlaying(_ player: AVAudioPlayer, successfully flag: Bool) {
        if flag {
            print("✅ Sound preview completed")
        } else {
            print("⚠️ Sound preview interrupted")
        }
        audioPlayer = nil
    }
}
```

- 合并了 `AudioPreviewService` 的播放功能

- 提供通知专用的 `UNNotificationSound` 获取方法

**文件**：`pethealthdata/pethealthdata/ViewModels/TodayViewModel.swift`（重构）

#### 3.4.2 生成规则

**代码结构（使用 @Observable）**：

```swift
import Foundation
import SwiftData

/// Today 视图模型（重构版 - 使用 @Observable）
@Observable
@MainActor
final class TodayViewModel {
    // MARK: - Properties
    
    var todayVaccines: [VaccineRecord] = []
    var todayMedications: [Medication] = []
    var upcomingReminders: [UpcomingReminder] = []
    var isLoading: Bool = false
    var errorMessage: String?
    var hasError: Bool = false
    
    private let modelContext: ModelContext
    private let calendar = Calendar.current
    
    // MARK: - Initialization
    
    init(modelContext: ModelContext) {
        self.modelContext = modelContext
    }
    
    // MARK: - Public Methods
    
    /// 加载今日提醒
    func loadTodaysReminders() {
        isLoading = true
        errorMessage = nil
        hasError = false
        
        do {
            let today = calendar.startOfDay(for: Date())
            let endOfToday = calendar.date(byAdding: .day, value: 1, to: today) ?? today
            
            // 1. 加载今日疫苗（使用谓词优化）
            try loadTodayVaccines(today: today, endOfToday: endOfToday)
            
            // 2. 加载今日药物（使用谓词优化）
            try loadTodayMedications(today: today)
            
            // 3. 加载即将到来提醒
            loadUpcomingReminders(today: today, days: 3)
            
            isLoading = false
        } catch {
            errorMessage = error.localizedDescription
            hasError = true
            isLoading = false
            print("❌ Failed to load today's reminders: \(error)")
        }
    }
    
    /// 标记药物为已服用
    func markMedicationAsTaken(_ medication: Medication, at time: Date) {
        medication.lastTaken = time
        try? modelContext.save()
    }
    
    /// 刷新数据
    func refresh() {
        loadTodaysReminders()
    }
    
    // MARK: - Private Methods
    
    /// 加载今日疫苗
    private func loadTodayVaccines(today: Date, endOfToday: Date) throws {
        // 使用谓词精确过滤
        let predicate = #Predicate<VaccineRecord> { vaccine in
            guard let nextDueDate = vaccine.nextDueDate else { return false }
            return nextDueDate >= today && nextDueDate < endOfToday
        }
        
        let descriptor = FetchDescriptor<VaccineRecord>(predicate: predicate)
        todayVaccines = try modelContext.fetch(descriptor)
        
        print("✅ Loaded \(todayVaccines.count) vaccines due today")
    }
    
    /// 加载今日药物
    private func loadTodayMedications(today: Date) throws {
        // 使用谓词过滤活跃药物
        let predicate = #Predicate<Medication> { medication in
            return medication.isActive && medication.startDate <= today
        }
        
        let descriptor = FetchDescriptor<Medication>(predicate: predicate)
        let allMedications = try modelContext.fetch(descriptor)
        
        // 在内存中进一步过滤（需要复杂逻辑）
        todayMedications = allMedications.filter { medication in
            // 1. 检查结束日期
            if let endDate = medication.endDate, endDate < today {
                return false
            }
            
            // 2. 检查是否已服用
            if let lastTaken = medication.lastTaken {
                let lastTakenDay = calendar.startOfDay(for: lastTaken)
                if lastTakenDay >= today {
                    return false
                }
            }
            
            // 3. 检查是否今天需要服用
            return medication.isDueToday(today: today)
        }
        
        print("✅ Loaded \(todayMedications.count) medications due today")
    }
    
    /// 加载即将到来的提醒
    private func loadUpcomingReminders(today: Date, days: Int) {
        let untilDate = calendar.date(byAdding: .day, value: days, to: today) ?? today
        var reminders: [UpcomingReminder] = []
        
        do {
            // 1. 即将到来的疫苗
            let vaccinePredicate = #Predicate<VaccineRecord> { vaccine in
                guard let nextDueDate = vaccine.nextDueDate else { return false }
                return nextDueDate > today && nextDueDate <= untilDate
            }
            
            let vaccineDescriptor = FetchDescriptor<VaccineRecord>(predicate: vaccinePredicate)
```

```swift
import SwiftUI
import AVFoundation

/// 通知声音选择器（重构版）
struct NotificationSoundPickerView: View {
    // MARK: - Properties
    
    @Binding var selectedSound: String
    @Environment(\.dismiss) private var dismiss
    
    private let sounds = NotificationSoundConfig.SoundOption.allCases
    @State private var previewPlaying: String?
    
    // MARK: - Body
    
    var body: some View {
        List {
            ForEach(sounds) { sound in
                SoundRow(
                    sound: sound,
                    isSelected: selectedSound == sound.fileName,
                    isPlaying: previewPlaying == sound.fileName,
                    onSelect: {
                        withAnimation(.easeInOut(duration: 0.2)) {
                            selectedSound = sound.fileName
                        }
                    },
                    onPreview: {
                        playPreview(sound: sound)
                    }
                )
            }
        }
        .navigationTitle("Notification Sound")
        .navigationBarTitleDisplayMode(.inline)
        .toolbar {
            ToolbarItem(placement: .confirmationAction) {
                Button("Done") {
                    dismiss()
                }
                .fontWeight(.semibold)
            }
        }
    }
    
    // MARK: - Private Methods
    
    private func playPreview(sound: NotificationSoundConfig.SoundOption) {
        guard !sound.fileName.isEmpty else {
            // Silent 选项
            return
        }
        
        previewPlaying = sound.fileName
        NotificationSoundValidator.shared.playPreview(fileName: sound.fileName) {
            // 播放完成
            withAnimation {
                previewPlaying = nil
            }
        }
    }
}

// MARK: - Sound Row Component

private struct SoundRow: View {
    let sound: NotificationSoundConfig.SoundOption
    let isSelected: Bool
    let isPlaying: Bool
    let onSelect: () -> Void
    let onPreview: () -> Void
    
    var body: some View {
        HStack(spacing: 12) {
            // 图标
            Image(systemName: isPlaying ? "speaker.wave.3.fill" : sound.previewIcon)
                .frame(width: 30)
                .foregroundColor(isPlaying ? .appPrimary : .secondary)
                .symbolEffect(.pulse, options: .repeating, value: isPlaying)
            
            // 名称
            Text(sound.name)
                .font(.system(size: 16))
                .foregroundColor(.primary)
            
            Spacer()
            
            // 选中标记
            if isSelected {
                Image(systemName: "checkmark.circle.fill")
                    .foregroundColor(.appPrimary)
                    .font(.system(size: 20))
            }
            
            // 预览按钮
            Button(action: onPreview) {
                Image(systemName: "speaker.wave.2.fill")
                    .foregroundColor(.appPrimary)
            }
            .disabled(sound.fileName.isEmpty)
            .opacity(sound.fileName.isEmpty ? 0.3 : 1.0)
        }
        .padding(.vertical, 8)
        .contentShape(Rectangle())
        .onTapGesture(perform: onSelect)
        .background(
            RoundedRectangle(cornerRadius: 10)
                .fill(isSelected ? Color.appPrimary.opacity(0.1) : Color.clear)
        )
    }
}

// MARK: - Preview

        NotificationSoundPickerView(selectedSound: .constant("triTone.caf"))
    }
}
```

#### 10.1.1 ViewModel 架构一致性

**问题**：
- 现有 `PetListViewModel` 和 `PetDetailViewModel` 使用 `@Observable` 宏（Swift 5.9）
- 当前 `TodayViewModel` 使用 `ObservableObject + @Published`（旧模式）

**修复方案**：
```swift
// ❌ 旧模式（当前实现）
@MainActor
class TodayViewModel: ObservableObject {
    @Published var todayVaccines: [VaccineRecord] = []
    @Published var todayMedications: [Medication] = []
    private var cancellables = Set<AnyCancellable>()
}

// ✅ 新模式（重构后）
@Observable
@MainActor
final class TodayViewModel {
    var todayVaccines: [VaccineRecord] = []
    var todayMedications: [Medication] = []
    // 不需要 @Published
    // 不需要 cancellables
}
```

**影响**：
- 删除 `import Combine`
- 删除 `cancellables` 属性
- 删除所有 `@Published` 标记
- 代码更简洁，符合 Swift 5.9+ 最佳实践

```swift
// ❌ 当前实现（创建独立 ModelContainer）
```

#### 10.1.3 isDueToday 逻辑审查

**问题**：
```swift
// ❌ 当前逻辑（比较今天开始时间）
if doseTime >= today {
    return true
}
```

**修复方案**：
```swift
// ✅ 修正逻辑（比较当前时间）
let now = Date()
if doseTime >= now {
    return true
}
```

**场景说明**：
- 如果现在是下午 3 点（15:00）
- 药物提醒时间是上午 9 点和下午 6 点
- 当前逻辑：显示需要服用（因为 9:00 和 18:00 都在今天）
- 修正逻辑：只显示下午 6 点的剂量（上午 9 点已过）
